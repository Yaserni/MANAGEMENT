pipeline {
  agent {
    docker {
      image 'windsekirun/jenkins-android-docker:1.1.1'
    }
  }

  options {
    // Stop the build early in case of compile or test failures
    skipStagesAfterUnstable()
  }
  stages {
    stage ('Prepare For Build'){
      steps {

        sh 'chmod +x ./gradlew'
      }
    }
    stage('Compile') {
      steps {
        // Compile the app
        sh './gradlew compileDebugSources'
      }
    }
    stage('Unit/Integration tests') {
      steps {
        // Compile and run the unit tests
        sh './gradlew testDebugUnitTest'
        sh 'find . -name "TEST-*.xml" -exec touch {} \\;'
        // Analyse the test results
        junit '**/TEST-*.xml'
      }
    }
    stage('Static Code Analysis') {
      steps {
	echo "Run Lint"
        sh './gradlew :app:lint'
	echo "run Sheck Style"
        sh './gradlew checkstyleMain'
      }
    }
    stage('Build APP') {
      steps {
	echo "App Building"
        sh './gradlew assembleDebug'
      }
    }
  }
post {
        always {
            recordIssues(
                tool: androidLintParser(pattern: 'app/build/reports/lint-results.xml'),
                unstableTotalAll: 140,
                failedTotalAll: 240
            )
            recordIssues(
                tool: checkStyle(pattern: 'app/build/reports/checkstyle/checkstyle-report.xml'),
                unstableTotalAll: 160,
                failedTotalAll: 330
            )
            
            echo 'Sending email.'
            
            emailext(
                to: 'yaserni@ac.sce.ac.il,abdalsk@ac.sce.ac.il,odaiwa@ac.sce.ac.il',
                body: "${currentBuild.currentResult}: Job ${env.JOB_NAME} build ${env.BUILD_NUMBER}\n More info at: ${env.BUILD_URL}",
                //recipientProviders: [[$class: 'DevelopersRecipientProvider'], [$class: 'RequesterRecipientProvider']],
                subject: "Jenkins Build ${currentBuild.currentResult}: Job ${env.JOB_NAME}"
            )
            
      }
    }
}
